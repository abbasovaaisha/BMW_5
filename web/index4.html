#include #include #define N 1000 // количество точек (≥ 1000) #define H 1e-5 // шаг дифференцирования // Простая тестовая функция double f(double x) { return x * x; } int main() { double a = 0.0, b = 1.0; double dx = (b - a) / (N - 1); // шаг по аргументу double deriv[N]; // массив для результатов // ---- Последовательный вариант ---- double start = omp_get_wtime(); for (int i = 0; i < N; i++) { double x = a + i * dx; deriv[i] = (f(x + H) - f(x - H)) / (2.0 * H); } double end = omp_get_wtime(); printf("Последовательное время: %f сек\n", end - start); // ---- Параллельный вариант (OpenMP) ---- start = omp_get_wtime(); #pragma omp parallel for for (int i = 0; i < N; i++) { double x = a + i * dx; deriv[i] = (f(x + H) - f(x - H)) / (2.0 * H); } end = omp_get_wtime(); printf("Параллельное время: %f сек\n", end - start); // Проверка одного значения (для контроля) int mid = N / 2; double x_mid = a + mid * dx; printf("Приближённое f'(%f) = %f (точное = %f)\n", x_mid, deriv[mid], 2.0 * x_mid); return 0; }
