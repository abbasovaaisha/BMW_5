#include #include #include #define N 1000 #define H 1e-5 using std::cout; using std::endl; using std::fixed; using std::setprecision; double f(double x) { return x * x; } int main() { double a = 0.0, b = 1.0; double dx = (b - a) / (N - 1); double deriv[N]; double start = omp_get_wtime(); for (int i = 0; i < N; ++i) { double x = a + i * dx; deriv[i] = (f(x + H) - f(x - H)) / (2.0 * H); } double end = omp_get_wtime(); cout << fixed << setprecision(6); cout << "Последовательное время: " << end - start << " сек\n"; start = omp_get_wtime(); #pragma omp parallel for for (int i = 0; i < N; ++i) { double x = a + i * dx; deriv[i] = (f(x + H) - f(x - H)) / (2.0 * H); } end = omp_get_wtime(); cout << "Параллельное время: " << end - start << " сек\n"; int mid = N / 2; double x_mid = a + mid * dx; cout << "Приближённое f'(" << x_mid << ") = " << deriv[mid] << " (точное = " << 2.0 * x_mid << ")\n"; return 0; }
